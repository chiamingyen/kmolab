#引入程式碼

~~~python
import math

# Pythagoras Theorem
def hypotenuse(side1, side2):
    return math.sqrt(math.pow(side1, 2)+math.pow(side2, 2))
    
side1 = 3
side2 = 4
h = hypotenuse(side1, side2)
print(a right triangle with side1=", side1, " and side2=", side2, "the hypotenuse is:", h)
~~~
